#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

char* solution(int a, int b) {
    // 리턴할 값은 메모리를 동적 할당해주세요.
    char* day_p[7] = { "SUN","MON","TUE","WED","THU","FRI","SAT" };
    int sumday = 4;
    int tmp;
    char ans;
    int last_day[12] = { 31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };

    for (int i = 0; i < a-1; i++) {
        sumday += last_day[i];
    }
    sumday += b;
    tmp = sumday % 7;

    char* answer = (char*)malloc(3);
    answer = day_p[tmp];
    return answer; 
}
